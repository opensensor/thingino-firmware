# Override u-boot default baseaddr for kernel 7.0. The 3.10 default 0x80600000
# only leaves 6 MiB of headroom before the compressed uImage; the 7.0 kernel
# decompresses to ~15 MiB and clobbers the still-compressed image, so bootm
# fails. 0x81800000 puts the compressed image 24 MiB above RAM start.
baseaddr=0x81800000
kern_addr=0x80000
kern_size=0x3F8000
